284  inline void next() {
285  if ( mPos[a] < mMax[a] ) {//by far this is the most common case
286  ++mPos[a];
287  } else if ( mPos[b] < mMax[b] ) {
288  mPos[a] = mMin[a];
289  ++mPos[b];

291  mPos[a] = mMin[a];

293  ++mPos[c];
294  }
